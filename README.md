def answer_one():
    import pandas as pd
    import numpy as np
    
    ''' 
         The comments were added by myself for reference and code readability in the future
         
         Read the Energy indicators excel file and save it into Energy Dataframe.
         we want to skip the header from excel file.If we go through the excel file, it can be noticed that 
         the first seventeen rows are unnecessary.
         
         Skip footer - we can calculate the number of rows to be skipped.last row no in excel is 283 
         the dataframe ends at 245 - so number of footer skip is 38. 
         we can use the pandas.read_excel function and pass the parameters to skip header and footer      
    '''
    Energy = pd.read_excel('Energy Indicators.xls', skiprows= 17, skipfooter=38)
    
    # Dropping unnecessary columns by their indeEnergy 
    
    Energy = Energy.drop(Energy.columns[[0,2]],axis=1)
    
    # Renaming the columns as per the excel sheet 
    
    Energy.columns = ['Country', 'Energy Supply', 'Energy Supply per Capita', '% Renewable']
    
    '''
        For all countries which have missing data (e.g. data with "...") 
        make sure this is reflected as np.NaN values. 
        Select the column indeEnergy and apply .to_numeric and coerce any errors encountered
        The series are in object type and .to_numeric will convert to float but strings that can't
        be converted to float 64 will be replaced automatically by NAN values
        another nice technique to was convert_objects but it got deprecated
    '''
    Energy[Energy.columns[[1,2,3]]] = Energy[Energy.columns[[1,2,3]]].apply(pd.to_numeric, errors='coerce')
    
    # Convert Energy Supply to gigajoules (there are 1,000,000 gigajoules in a petajoule).
    
    Energy['Energy Supply'] = Energy['Energy Supply']*1000000
    # Replace the country names as per the requirement 
    
    Energy['Country'] = Energy['Country'].replace({'China, Hong Kong Special Administrative Region':'Hong Kong',
                                     'United Kingdom of Great Britain and Northern Ireland':'United Kingdom',
                                     'Republic of Korea':'South Korea','United States of America':'United States',
                                     'Iran (Islamic Republic of)':'Iran'})
    
    # There are also several countries with numbers and/or parenthesis in their name. 
    # We use pandas series string handling functions with regeEnergy patterns
    Energy['Country'] = Energy['Country'].str.replace(" \(.*\)","")
   
    
    # Read the GDP data from world bank csv and skip the header and save into GDP dataframe
    
    GDP = pd.read_csv('world_bank.csv', skiprows=4)
    
    #Replace the country name as per the requirement 
    GDP['Country Name'] = GDP['Country Name'].replace({"Korea, Rep.": "South Korea", 
                                                       "Iran, Islamic Rep.": "Iran",
                                                       "Hong Kong SAR, China": "Hong Kong"})
    
    #Select the columns eEnergyplaining the last ten years of GDP data only
    
    GDP = GDP[['Country Name','2006','2007','2008','2009','2010','2011','2012','2013','2014','2015']]
    
    GDP.rename(columns={'Country Name': 'Country'}, inplace=True)
    
    
    # Read the Sciamgo Journal and Country Rank data for Energy Engineering and Power Technology from scimagojr-3.EnergylsEnergy 
    # It ranks countries based on their journal contributions in the aforementioned area
    
    ScimEn = pd.read_excel('scimagojr-3.xlsx')
    
    ScimEn_top15 = ScimEn[:15]
    
    '''
        Join the three datasets: GDP, Energy, and ScimEn into a new dataset (using intersection of country names). 
        Use only the last 10 years of GDP data and only top 15 countries by Scimagojr 'Rank'.

        The indeEnergy of this DataFrame should be the name of the country 
        The columns - ['Rank', 'Documents', 'Citable documents', 'Citations', 'Self-citations', 
        'Citations per document', 'H indeEnergy', 'Energy Supply', 'Energy Supply per Capita', '% Renewable',
        '2006', '2007', '2008', '2009', '2010', '2011', '2012', '2013', '2014', '2015'].
    '''
    
    df = pd.merge(ScimEn_top15, Energy, how = 'inner', left_on = 'Country', right_on='Country')
    ans_df = pd.merge(df,GDP, how = 'inner', left_on = 'Country', right_on='Country')
    ans_df = ans_df.set_index('Country')
    
    
    
    return ans_df
answer_one()
